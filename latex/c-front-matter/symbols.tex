\chapter*{Symbols}
\addcontentsline{toc}{chapter}{Symbols}

\noindent
\begin{tabular}{ll}
$N$ & Number of candidates \\
$V$ & Number of cast votes \\
$\mathit{candidateId}$ & Chosen candidate index \\
$\mathit{salt}$ & Random salt used in commitment \\
$\mathit{commitment}$ & Ballot commitment, $\mathsf{Poseidon}(\mathit{candidateId},\mathit{salt})$ \\
$H_{ct}$ & Ciphertext binding hash \\
$\mathit{PK}$ & ElGamal public key \\
$\mathit{sk}$ & ElGamal secret key \\
$G$ & BabyJubJub subgroup generator \\
$C1, C2$ & ElGamal ciphertext points \\
$\mathit{merkleRoot}$ & Merkle root of all commitments \\
\end{tabular}

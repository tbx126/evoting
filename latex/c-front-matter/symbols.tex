\chapter*{Symbols}
\addcontentsline{toc}{chapter}{Symbols}

\noindent
\begin{tabular}{ll}
$N$ & Number of candidates \\
$V$ & Number of cast votes \\
$v$ & Chosen candidate index \\
$s$ & Random salt used in commitment \\
$Com$ & Ballot commitment, $Poseidon(v,s)$ \\
$H_{ct}$ & Ciphertext binding hash \\
$pk$ & ElGamal public key \\
$sk$ & ElGamal secret key \\
$C1, C2$ & ElGamal ciphertext points \\
$R$ & Merkle root of all commitments \\
\end{tabular}
